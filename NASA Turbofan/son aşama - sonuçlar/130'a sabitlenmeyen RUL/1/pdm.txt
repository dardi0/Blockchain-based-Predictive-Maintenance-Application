import pandas as pd
import numpy as np
from sklearn.preprocessing import MinMaxScaler, RobustScaler
import matplotlib.pyplot as plt
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import LSTM, Dense, Conv1D, MaxPooling1D, Dropout, BatchNormalization
from sklearn.model_selection import train_test_split
from sklearn.metrics import mean_squared_error, r2_score, precision_score, recall_score, f1_score, accuracy_score
import math
import tensorflow as tf

def load_data():
    """Train, test ve RUL veri setlerini yükle"""
    # Train veri setini yükle
    train_df = pd.read_csv('turbofan/train_FD001.txt', sep=' ', header=None)
    train_df = train_df.drop(columns=[26, 27])
    
    # Test veri setini yükle
    test_df = pd.read_csv('turbofan/test_FD001.txt', sep=' ', header=None)
    test_df = test_df.drop(columns=[26, 27])
    
    # RUL veri setini yükle
    rul_df = pd.read_csv('turbofan/RUL_FD001.txt', header=None)
    rul_df.columns = ['RUL']
    
    # Sütun isimlerini belirleme
    columns = ['unit', 'cycle', 'op1', 'op2', 'op3'] + [f's{i}' for i in range(1,22)]
    train_df.columns = columns
    test_df.columns = columns
    
    # Kullanılacak özellikler 
    # Operating conditions: op1, op2
    # Sensörler: 2-4, 6-9, 11-14, 16-18, 20-21
    selected_sensors = [2, 3, 4, 6, 7, 8, 10, 11, 12, 13, 15, 16, 17, 18, 20, 21]
    features_to_use = ['op1', 'op2'] + [f's{i}' for i in selected_sensors]

    return train_df, test_df, rul_df, features_to_use  


def prepare_sequences(df, sequence_length=30, selected_sensors=None, is_test=False, sensor_scaler=None, op_scaler=None):
    """
    Veriyi LSTM için sequence'lere dönüştürür.
    
    Eğer sensor_scaler ve op_scaler sağlanmışsa, bunları kullanarak transform yapar.
    Sağlanmamışsa, eğitim verisi için fit edip transform eder.
    """
    if selected_sensors is None:
        feature_cols = [f's{i}' for i in range(1,22)]
    else:
        feature_cols = selected_sensors
    
    # Sensör verilerinin ölçeklendirilmesi:
    if sensor_scaler is None:
        sensor_scaler = RobustScaler()
        df[feature_cols] = sensor_scaler.fit_transform(df[feature_cols])
    else:
        df[feature_cols] = sensor_scaler.transform(df[feature_cols])
    
    # Operating conditions (op1, op2) için ölçeklendirme:
    op_cols = ['op1', 'op2']
    if op_scaler is None:
        op_scaler = MinMaxScaler()
        df[op_cols] = op_scaler.fit_transform(df[op_cols])
    else:
        df[op_cols] = op_scaler.transform(df[op_cols])
    
    # Tüm özellikler: sensörler ve op'lar
    full_feature_cols = feature_cols + op_cols
    
    sequences = []
    targets = []
    
    for unit in df['unit'].unique():
        unit_data = df[df['unit'] == unit]
        for i in range(len(unit_data) - sequence_length):
            sequence = unit_data[full_feature_cols].iloc[i:(i + sequence_length)].values
            if not is_test:
                target = unit_data['RUL'].iloc[i + sequence_length]
                targets.append(target)
            sequences.append(sequence)
    
    if is_test:

        return np.array(sequences), None, sensor_scaler, op_scaler
    return np.array(sequences), np.array(targets), sensor_scaler, op_scaler

def hybrid_model(sequence_length, n_features):
    model = Sequential([
        # CNN katmanları
        Conv1D(128, 5, activation='relu', padding='same', input_shape=(sequence_length, n_features)),
        BatchNormalization(),
        MaxPooling1D(3),
        Dropout(0.3),
        
        Conv1D(64, 3, activation='relu', padding='same'),
        BatchNormalization(),
        MaxPooling1D(2),
        Dropout(0.2),
        
        # LSTM katmanları
        LSTM(100, return_sequences=True, activation = "tanh"),
        BatchNormalization(),
        Dropout(0.2),
        
        LSTM(50, return_sequences = False, activation = "tanh"),
        BatchNormalization(),
        Dropout(0.2),
        
        # Çıkış katmanları
        Dense(50, activation='relu'),
        BatchNormalization(),
        Dropout(0.3),
        Dense(20, activation = "relu"),
        Dense(1, activation='linear')
    ])
    
    optimizer = tf.keras.optimizers.Adam(learning_rate=0.001)
    model.compile(optimizer=optimizer, loss='mse', metrics=['mae'])
    return model

# Verileri yükle
train_df, test_df, rul_df, features_to_use = load_data()

# features_to_use içinden sensör ve işletim parametrelerini ayır
op_cols = [f for f in features_to_use if f.startswith('op')]
selected_sensors = [f for f in features_to_use if f.startswith('s')]

print("\nKullanılacak özellikler:")
print(f"İşletim parametreleri: {op_cols}")
print(f"Sensörler: {selected_sensors}")
print(f"Toplam sensör sayısı: {len(selected_sensors)}")
print(f"Toplam işletim parametresi sayısı: {len(op_cols)}")
print(f"Toplam özellik sayısı: {len(features_to_use)}")

# RUL hesaplama için parametre tanımları
sequence_length = 50    # Pencere uzunluğu
max_rul_value = 130     # ESKİ: Regresyon için maksimum RUL değeri (üst sınır) - artık KULLANILMIYOR
rul_threshold = 50      # Sınıflandırma için eşik değeri (RUL > 50 ise 1, değilse 0)

print("\nRUL hesaplama parametreleri:")
print(f"- Sabit RUL pencere uzunluğu: {sequence_length} döngü")
print(f"- Sınıflandırma eşiği: {rul_threshold} (RUL > {rul_threshold} ise 1, değilse 0)")
print("- ARTIK RUL değerleri sınırlandırılmıyor, gerçek değerler kullanılıyor")

# Eğitim verisi için RUL hesaplama
for unit in train_df['unit'].unique():
    unit_data = train_df[train_df['unit'] == unit]
    max_cycle = unit_data['cycle'].max()
    
    # ESKİ YAKLAŞIM:
    # Her cycle için RUL değeri hesapla
    # for idx, row in unit_data.iterrows():
    #     current_cycle = row['cycle']
    #     # İlk sequence_length cycle için sabit RUL değeri
    #     if current_cycle <= sequence_length:
    #         train_df.loc[idx, 'RUL'] = max_rul_value
    #     else:
    #         # sequence_length sonrası için geri kalan cycle sayısı
    #         raw_rul = max_cycle - current_cycle
    #         # Regresyon için 130 üst sınırını uygula
    #         train_df.loc[idx, 'RUL'] = min(raw_rul, max_rul_value)
    
    # YENİ YAKLAŞIM: Her cycle için gerçek RUL değerini hesapla
    train_df.loc[train_df['unit'] == unit, 'RUL'] = max_cycle - train_df.loc[train_df['unit'] == unit, 'cycle']
    
    # Sınıflandırma etiketi ekle
    train_df.loc[train_df['unit'] == unit, 'RUL_binary'] = (train_df.loc[train_df['unit'] == unit, 'RUL'] > rul_threshold).astype(int)

# Test verisi için RUL hesaplama
print("\nTest verisi için RUL hesaplanıyor...")
# Her motor için maksimum döngüyü ve RUL_FD001.txt'deki karşılık gelen değeri kullanarak hesaplayalım
for i, unit in enumerate(test_df['unit'].unique()):
    unit_data = test_df[test_df['unit'] == unit]
    max_cycle = unit_data['cycle'].max()
    # RUL dosyasından ilgili motor için kalan ömür değerini al
    remaining_rul = rul_df.iloc[i]['RUL']
    
    # ESKİ YAKLAŞIM:
    # Her cycle için RUL değeri hesapla
    # for idx, row in unit_data.iterrows():
    #     current_cycle = row['cycle']
    #     raw_rul = max_cycle + remaining_rul - current_cycle
    #     test_df.loc[idx, 'RUL'] = min(raw_rul, max_rul_value)
    
    # YENİ YAKLAŞIM: Gerçek RUL değerleri
    test_df.loc[test_df['unit'] == unit, 'RUL'] = max_cycle + remaining_rul - test_df.loc[test_df['unit'] == unit, 'cycle']
    
    # Sınıflandırma etiketi ekle
    test_df.loc[test_df['unit'] == unit, 'RUL_binary'] = (test_df.loc[test_df['unit'] == unit, 'RUL'] > rul_threshold).astype(int)

print(f"Eğitim verisinde toplam {len(train_df['unit'].unique())} motor bulunuyor.")
print(f"Test verisinde toplam {len(test_df['unit'].unique())} motor bulunuyor.")

# Seçilen sensörlerle sequence'leri hazırla (eğitim verisi için scaler'ı fit et)
X, y, sensor_scaler, op_scaler = prepare_sequences(train_df, selected_sensors=selected_sensors)

# --- RUL değerlerini de ölçekleyelim ---
rul_scaler = MinMaxScaler()
y_scaled = rul_scaler.fit_transform(y.reshape(-1, 1)).flatten()

# Eğitim ve test verilerini ayır (y_scaled kullanıyoruz)
X_train, X_test, y_train, y_test = train_test_split(X, y_scaled, test_size=0.2, random_state=42)

# Model parametrelerini güncelle: n_features = (seçilmiş sensörler) + (op1, op2)
n_features = len(selected_sensors) + 2

# Modeli oluştur
model = hybrid_model(30, n_features)
print("\nModel Özeti:")
model.summary()

# Özelleştirilmiş Early Stopping
early_stopping = tf.keras.callbacks.EarlyStopping(
    monitor='val_loss',
    patience=35,
    min_delta=0.001,
    mode='min',
    restore_best_weights=True
)

reduce_lr = tf.keras.callbacks.ReduceLROnPlateau(
    monitor='val_loss',
    factor=0.1,
    patience=10,
    min_lr=1e-5
)

# Modeli eğit
history = model.fit(
    X_train, y_train,
    epochs=300,
    batch_size=64,
    validation_split=0.2,
    callbacks=[early_stopping, reduce_lr],
    verbose=1
)

# Test verisi için sequence'leri oluştururken,
# Eğitimde fit ettiğimiz sensor_scaler ve op_scaler'ı kullanıyoruz.
X_test_seq, _, _, _ = prepare_sequences(test_df, sequence_length=30, selected_sensors=selected_sensors, 
                                          is_test=True, sensor_scaler=sensor_scaler, op_scaler=op_scaler)

# Test verisi üzerinde tahmin yap
y_pred = model.predict(X_test_seq)

# Tahminleri orijinal ölçeğe geri dönüştür
y_pred = rul_scaler.inverse_transform(y_pred)

# Her ünite için sadece son cycle'daki RUL tahminini al
final_predictions = []
final_true_values = []

current_idx = 0
for unit in test_df['unit'].unique():
    unit_mask = test_df['unit'] == unit
    unit_data = test_df[unit_mask]
    sequence_count = len(unit_data) - 30  # sequence_length = 30
    
    # Son sequence'in tahminini al
    if sequence_count > 0:
        last_sequence_idx = current_idx + sequence_count - 1
        final_predictions.append(float(y_pred[last_sequence_idx][0]))
        final_true_values.append(float(rul_df.iloc[int(unit)-1]['RUL']))
    
    current_idx += sequence_count

# Metrikleri hesapla
final_predictions = np.array(final_predictions)
final_true_values = np.array(final_true_values)

# --- Regresyon Performans Metrikleri ---
# MSE (Mean Squared Error - Ortalama Kare Hata)
# Tahmin hatalarının karelerinin ortalaması. Büyük hataları daha fazla cezalandırır.
mse = mean_squared_error(final_true_values, final_predictions)

# RMSE (Root Mean Squared Error - Kök Ortalama Kare Hata)
# MSE'nin kare kökü. Sonuç, orijinal veri ile aynı birimde olduğu için daha yorumlanabilir.
rmse = math.sqrt(mse)

# MAE (Mean Absolute Error - Ortalama Mutlak Hata)
# Tahmin hatalarının mutlak değerlerinin ortalaması. Aykırı değerlere MSE kadar duyarlı değildir.
mae = np.mean(np.abs(final_true_values - final_predictions))

# MAPE (Mean Absolute Percentage Error - Ortalama Mutlak Yüzde Hata)
# Tahmin hatalarının gerçek değere oranının yüzde olarak ortalaması. Göreli hata ölçüsüdür.
mape = np.mean(np.abs((final_true_values - final_predictions) / final_true_values)) * 100

# R² (Determination Coefficient - Belirtme Katsayısı)
# Modelin veri değişkenliğini açıklama yeteneği. 1'e yakın değerler daha iyi uyumu gösterir.
r2 = r2_score(final_true_values, final_predictions)

print("\nModel Performansı (Regresyon Metrikleri):")
print(f"MSE: {mse:.4f}")
print(f"RMSE: {rmse:.4f}")
print(f"MAE: {mae:.4f}")
print(f"MAPE: {mape:.4f}%")
print(f"R²: {r2:.4f}")

# --- İkili Sınıflandırma Metrikleri ---
# RUL için eşik değeri belirle (50 cycle ve altındaki değerler Dejenere kabul edilecek)
rul_threshold = 50

# Tahminleri ve gerçek değerleri ikili sınıflandırmaya dönüştür
# RUL > 50 ise 1 (Dejenere olmayan/Sağlıklı), RUL <= 50 ise 0 (Dejenere)
y_pred_binary = (final_predictions > rul_threshold).astype(int)
y_true_binary = (final_true_values > rul_threshold).astype(int)

# Sınıflandırma metriklerini hesapla
accuracy = accuracy_score(y_true_binary, y_pred_binary)
precision = precision_score(y_true_binary, y_pred_binary, zero_division=0)
recall = recall_score(y_true_binary, y_pred_binary, zero_division=0)
f1 = f1_score(y_true_binary, y_pred_binary, zero_division=0)

print(f"\nİkili Sınıflandırma Metrikleri (RUL > {rul_threshold} eşiği için):")
print(f"Accuracy: {accuracy:.4f}")
print(f"Precision: {precision:.4f} (Sağlıklı olarak tahmin edilenlerin gerçekten Sağlıklı olma oranı)")
print(f"Recall: {recall:.4f} (Gerçekte Sağlıklı olanların doğru tespit edilme oranı)")
print(f"F1-Score: {f1:.4f}")
